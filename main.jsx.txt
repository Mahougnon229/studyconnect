import React, { useState, useEffect, useRef } from 'react';
import { 
  Users, MessageCircle, Phone, User, Settings, Bell, Lock, 
  Palette, HelpCircle, LogOut, Home, Video, Send, 
  Image as ImageIcon, Smile, Search, MoreVertical, 
  ThumbsUp, Heart, Laugh, AlertCircle, 
  Plus, X, Edit, Trash2, Moon, Sun,
  ChevronLeft, Upload, Check, Camera, Mic, MicOff,
  VideoOff, PhoneOff, Menu, Clock, MapPin
} from 'lucide-react';

// Stockage persistant
const storage = window.storage;

// Logo StudyConnect

  const dimensions = size === "large" ? { width: 200, height: 75 } : { width: 120, height: 45 };
  
  return (
    <svg width={dimensions.width} height={dimensions.height} viewBox="0 0 800 300" style={{ display: 'block' }}>
      <circle cx="80" cy="150" r="60" fill="#2962FF" />
      <circle cx="220" cy="150" r="60" fill="#6495ED" />
      <line x1="140" y1="150" x2="160" y2="150" stroke="#2962FF" strokeWidth="8" />
      <text x="80" y="175" textAnchor="middle" fill="white" fontSize="80" fontWeight="bold" fontFamily="Arial, sans-serif">S</text>
      <text x="220" y="175" textAnchor="middle" fill="white" fontSize="80" fontWeight="bold" fontFamily="Arial, sans-serif">C</text>
      <text x="320" y="135" fill="#2962FF" fontSize="60" fontWeight="bold" fontFamily="Arial, sans-serif">Study</text>
      <text x="320" y="195" fill="#6495ED" fontSize="60" fontWeight="bold" fontFamily="Arial, sans-serif">Connect</text>
    </svg>
  );
};

// Composant Emoji Picker simple
const EmojiPicker = ({ onSelect, onClose }) => {
  const emojis = ['😀', '😂', '❤️', '👍', '🎉', '🔥', '💯', '✨', '👏', '🙌', '💪', '🎓', '📚', '✍️', '💡', '🤔', '😊', '😍', '🥳', '🤗'];
  
  return (
    <div style={{
      position: 'absolute',
      bottom: '100%',
      left: 0,
      background: 'white',
      border: '1px solid #e5e7eb',
      borderRadius: '12px',
      padding: '12px',
      display: 'grid',
      gridTemplateColumns: 'repeat(5, 1fr)',
      gap: '8px',
      boxShadow: '0 4px 12px rgba(0,0,0,0.1)',
      zIndex: 1000,
      marginBottom: '8px'
    }}>
      {emojis.map(emoji => (
        <button
          key={emoji}
          onClick={() => {
            onSelect(emoji);
            onClose();
          }}
          style={{
            background: 'transparent',
            border: 'none',
            fontSize: '24px',
            cursor: 'pointer',
            padding: '8px',
            borderRadius: '6px',
            transition: 'background 0.2s'
          }}
          onMouseOver={(e) => e.target.style.background = '#f3f4f6'}
          onMouseOut={(e) => e.target.style.background = 'transparent'}
        >
          {emoji}
        </button>
      ))}
    </div>
  );
};

// Composant principal
export default function StudyConnect() {
  const [screen, setScreen] = useState('welcome');
  const [currentUser, setCurrentUser] = useState(null);
  const [darkMode, setDarkMode] = useState(false);
  const [activeTab, setActiveTab] = useState('feed');
  
  // États pour l'inscription
  const [signupForm, setSignupForm] = useState({ nom: '', prenom: '', specialite: '' });
  
  // États pour les fonctionnalités
  const [posts, setPosts] = useState([]);
  const [newPost, setNewPost] = useState('');
  const [conversations, setConversations] = useState([]);
  const [selectedConversation, setSelectedConversation] = useState(null);
  const [newMessage, setNewMessage] = useState('');
  const [showEmojiPicker, setShowEmojiPicker] = useState(false);
  const [notifications, setNotifications] = useState(true);
  const [onlineStatus, setOnlineStatus] = useState('visible');
  const [showGuide, setShowGuide] = useState(false);
  const [showAbout, setShowAbout] = useState(false);
  const [userBio, setUserBio] = useState('');
  const [editingBio, setEditingBio] = useState(false);
  const [allUsers, setAllUsers] = useState([]);
  
  // États pour les appels
  const [inCall, setInCall] = useState(false);
  const [callType, setCallType] = useState('video'); // 'audio' ou 'video'
  const [isMuted, setIsMuted] = useState(false);
  const [isVideoOff, setIsVideoOff] = useState(false);
  const [callWith, setCallWith] = useState(null);
  const videoRef = useRef(null);
  const [stream, setStream] = useState(null);
  
  // Charger les données au démarrage
  useEffect(() => {
    loadData();
  }, []);
  
  const loadData = async () => {
    try {
      // Charger les posts
      const postsData = await storage.get('posts');
      if (postsData) {
        setPosts(JSON.parse(postsData.value));
      }
      
      // Charger les conversations
      const convsData = await storage.get('conversations');
      if (convsData) {
        setConversations(JSON.parse(convsData.value));
      }
      
      // Charger tous les utilisateurs
      const usersData = await storage.get('all_users');
      if (usersData) {
        setAllUsers(JSON.parse(usersData.value));
      }
    } catch (error) {
      console.log('Première initialisation');
    }
  };
  
  // Fonction d'inscription
  const handleSignup = async () => {
    if (!signupForm.nom.trim() || !signupForm.prenom.trim() || !signupForm.specialite.trim()) {
      alert('Veuillez remplir tous les champs');
      return;
    }
    
    const user = {
      id: Date.now(),
      nom: signupForm.nom.trim(),
      prenom: signupForm.prenom.trim(),
      specialite: signupForm.specialite.trim().toUpperCase(),
      dateInscription: new Date().toISOString(),
      photo: null,
      bio: '',
      online: true
    };
    
    // Ajouter l'utilisateur à la liste globale
    const updatedUsers = [...allUsers, user];
    setAllUsers(updatedUsers);
    await storage.set('all_users', JSON.stringify(updatedUsers));
    
    setCurrentUser(user);
    setScreen('app');
  };
  
  // Créer un post
  const createPost = async () => {
    if (!newPost.trim()) return;
    
    const post = {
      id: Date.now(),
      author: currentUser,
      content: newPost,
      date: new Date().toISOString(),
      reactions: { like: 0, love: 0, laugh: 0, sad: 0, wow: 0 },
      comments: []
    };
    
    const updatedPosts = [post, ...posts];
    setPosts(updatedPosts);
    await storage.set('posts', JSON.stringify(updatedPosts));
    setNewPost('');
  };
  
  // Réagir à un post
  const reactToPost = async (postId, reaction) => {
    const updatedPosts = posts.map(post => {
      if (post.id === postId) {
        return {
          ...post,
          reactions: {
            ...post.reactions,
            [reaction]: post.reactions[reaction] + 1
          }
        };
      }
      return post;
    });
    
    setPosts(updatedPosts);
    await storage.set('posts', JSON.stringify(updatedPosts));
  };
  
  // Créer une nouvelle conversation
  const createConversation = (otherUser) => {
    const existingConv = conversations.find(c => 
      c.participants.some(p => p.id === otherUser.id)
    );
    
    if (existingConv) {
      setSelectedConversation(existingConv);
      return;
    }
    
    const newConv = {
      id: Date.now(),
      participants: [currentUser, otherUser],
      messages: [],
      lastMessage: null
    };
    
    const updatedConvs = [...conversations, newConv];
    setConversations(updatedConvs);
    storage.set('conversations', JSON.stringify(updatedConvs));
    setSelectedConversation(newConv);
  };
  
  // Envoyer un message
  const sendMessage = async () => {
    if (!newMessage.trim() || !selectedConversation) return;
    
    const message = {
      id: Date.now(),
      sender: currentUser,
      content: newMessage,
      date: new Date().toISOString(),
      read: false
    };
    
    const updatedConvs = conversations.map(conv => {
      if (conv.id === selectedConversation.id) {
        return {
          ...conv,
          messages: [...conv.messages, message],
          lastMessage: message
        };
      }
      return conv;
    });
    
    setConversations(updatedConvs);
    setSelectedConversation({
      ...selectedConversation,
      messages: [...selectedConversation.messages, message]
    });
    await storage.set('conversations', JSON.stringify(updatedConvs));
    setNewMessage('');
  };
  
  // Démarrer un appel
  const startCall = async (user, type) => {
    try {
      const constraints = {
        audio: true,
        video: type === 'video'
      };
      
      const mediaStream = await navigator.mediaDevices.getUserMedia(constraints);
      setStream(mediaStream);
      
      if (videoRef.current && type === 'video') {
        videoRef.current.srcObject = mediaStream;
      }
      
      setCallWith(user);
      setCallType(type);
      setInCall(true);
    } catch (error) {
      alert('Impossible d\'accéder à la caméra/micro. Vérifiez les autorisations.');
      console.error('Erreur média:', error);
    }
  };
  
  // Terminer un appel
  const endCall = () => {
    if (stream) {
      stream.getTracks().forEach(track => track.stop());
    }
    setInCall(false);
    setCallWith(null);
    setStream(null);
    setIsMuted(false);
    setIsVideoOff(false);
  };
  
  // Toggle micro
  const toggleMute = () => {
    if (stream) {
      stream.getAudioTracks().forEach(track => {
        track.enabled = !track.enabled;
      });
      setIsMuted(!isMuted);
    }
  };
  
  // Toggle vidéo
  const toggleVideo = () => {
    if (stream && callType === 'video') {
      stream.getVideoTracks().forEach(track => {
        track.enabled = !track.enabled;
      });
      setIsVideoOff(!isVideoOff);
    }
  };
  
  // Mettre à jour la bio
  const updateBio = async () => {
    const updatedUser = { ...currentUser, bio: userBio };
    setCurrentUser(updatedUser);
    
    const updatedUsers = allUsers.map(u => 
      u.id === currentUser.id ? updatedUser : u
    );
    setAllUsers(updatedUsers);
    await storage.set('all_users', JSON.stringify(updatedUsers));
    setEditingBio(false);
  };
  
  // Styles
  const theme = {
    primary: '#2962FF',
    secondary: '#6495ED',
    background: darkMode ? '#1a1a1a' : '#f5f7fa',
    surface: darkMode ? '#2d2d2d' : '#ffffff',
    text: darkMode ? '#ffffff' : '#1a1a1a',
    textSecondary: darkMode ? '#b0b0b0' : '#6b7280',
    border: darkMode ? '#404040' : '#e5e7eb'
  };
  
  // Écran de bienvenue
  if (screen === 'welcome') {
    return (
      <div style={{
        minHeight: '100vh',
        background: `linear-gradient(135deg, ${theme.primary} 0%, ${theme.secondary} 100%)`,
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        padding: '20px',
        fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif'
      }}>
        <div style={{
          background: 'white',
          borderRadius: '24px',
          padding: '48px',
          maxWidth: '500px',
          width: '100%',
          boxShadow: '0 20px 60px rgba(0,0,0,0.3)',
          textAlign: 'center'
        }}>
          <div style={{ marginBottom: '32px' }}>
            <Logo size="large" />
          </div>
          
          <h1 style={{
            fontSize: '32px',
            fontWeight: 'bold',
            color: theme.primary,
            marginBottom: '16px'
          }}>
            Bienvenue sur StudyConnect
          </h1>
          
          <p style={{
            fontSize: '18px',
            color: '#6b7280',
            marginBottom: '40px'
          }}>
            Votre espace d'échange entre étudiants
          </p>
          
          <button
            onClick={() => setScreen('signup')}
            style={{
              background: theme.primary,
              color: 'white',
              padding: '16px 48px',
              borderRadius: '12px',
              border: 'none',
              fontSize: '16px',
              fontWeight: '600',
              cursor: 'pointer',
              transition: 'all 0.2s',
              width: '100%'
            }}
            onMouseOver={(e) => e.target.style.transform = 'scale(1.02)'}
            onMouseOut={(e) => e.target.style.transform = 'scale(1)'}
          >
            Commencer
          </button>
        </div>
      </div>
    );
  }
  
  // Écran d'inscription
  if (screen === 'signup') {
    return (
      <div style={{
        minHeight: '100vh',
        background: `linear-gradient(135deg, ${theme.primary} 0%, ${theme.secondary} 100%)`,
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        padding: '20px',
        fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif'
      }}>
        <div style={{
          background: 'white',
          borderRadius: '24px',
          padding: '48px',
          maxWidth: '500px',
          width: '100%',
          boxShadow: '0 20px 60px rgba(0,0,0,0.3)'
        }}>
          <button
            onClick={() => setScreen('welcome')}
            style={{
              background: 'transparent',
              border: 'none',
              color: theme.primary,
              cursor: 'pointer',
              marginBottom: '24px',
              display: 'flex',
              alignItems: 'center',
              gap: '8px',
              fontSize: '14px'
            }}
          >
            <ChevronLeft size={20} /> Retour
          </button>
          
          <div style={{ marginBottom: '32px', textAlign: 'center' }}>
            <Logo size="small" />
          </div>
          
          <h2 style={{
            fontSize: '28px',
            fontWeight: 'bold',
            color: '#1a1a1a',
            marginBottom: '8px',
            textAlign: 'center'
          }}>
            Créer mon profil
          </h2>
          
          <p style={{
            fontSize: '16px',
            color: '#6b7280',
            marginBottom: '32px',
            textAlign: 'center'
          }}>
            Entrez vos informations pour commencer
          </p>
          
          <div style={{ display: 'flex', flexDirection: 'column', gap: '20px' }}>
            <div>
              <label style={{ display: 'block', marginBottom: '8px', fontSize: '14px', fontWeight: '600' }}>
                Nom
              </label>
              <input
                type="text"
                value={signupForm.nom}
                onChange={(e) => setSignupForm({...signupForm, nom: e.target.value})}
                placeholder="Votre nom"
                style={{
                  width: '100%',
                  padding: '14px',
                  borderRadius: '8px',
                  border: '1px solid #e5e7eb',
                  fontSize: '16px',
                  boxSizing: 'border-box'
                }}
              />
            </div>
            
            <div>
              <label style={{ display: 'block', marginBottom: '8px', fontSize: '14px', fontWeight: '600' }}>
                Prénom
              </label>
              <input
                type="text"
                value={signupForm.prenom}
                onChange={(e) => setSignupForm({...signupForm, prenom: e.target.value})}
                placeholder="Votre prénom"
                style={{
                  width: '100%',
                  padding: '14px',
                  borderRadius: '8px',
                  border: '1px solid #e5e7eb',
                  fontSize: '16px',
                  boxSizing: 'border-box'
                }}
              />
            </div>
            
            <div>
              <label style={{ display: 'block', marginBottom: '8px', fontSize: '14px', fontWeight: '600' }}>
                Spécialité
              </label>
              <input
                type="text"
                value={signupForm.specialite}
                onChange={(e) => setSignupForm({...signupForm, specialite: e.target.value})}
                placeholder="Ex: SD, TL, EGP, AUDIT"
                style={{
                  width: '100%',
                  padding: '14px',
                  borderRadius: '8px',
                  border: '1px solid #e5e7eb',
                  fontSize: '16px',
                  boxSizing: 'border-box'
                }}
              />
            </div>
            
            <button
              onClick={handleSignup}
              style={{
                background: theme.primary,
                color: 'white',
                padding: '16px',
                borderRadius: '12px',
                border: 'none',
                fontSize: '16px',
                fontWeight: '600',
                cursor: 'pointer',
                transition: 'all 0.2s',
                marginTop: '8px'
              }}
            >
              Entrer dans StudyConnect
            </button>
          </div>
        </div>
      </div>
    );
  }
  
  // Interface d'appel
  if (inCall) {
    return (
      <div style={{
        minHeight: '100vh',
        background: '#000',
        display: 'flex',
        flexDirection: 'column',
        fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif'
      }}>
        {/* Vidéo */}
        <div style={{ flex: 1, position: 'relative', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
          {callType === 'video' && !isVideoOff ? (
            <video
              ref={videoRef}
              autoPlay
              muted
              playsInline
              style={{
                width: '100%',
                height: '100%',
                objectFit: 'cover'
              }}
            />
          ) : (
            <div style={{
              display: 'flex',
              flexDirection: 'column',
              alignItems: 'center',
              gap: '24px'
            }}>
              <div style={{
                width: '120px',
                height: '120px',
                borderRadius: '50%',
                background: theme.primary,
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                color: 'white',
                fontSize: '48px',
                fontWeight: 'bold'
              }}>
                {callWith?.prenom[0]}{callWith?.nom[0]}
              </div>
              <div style={{ textAlign: 'center', color: 'white' }}>
                <h2 style={{ fontSize: '24px', marginBottom: '8px' }}>
                  {callWith?.prenom} {callWith?.nom}
                </h2>
                <p style={{ color: '#9ca3af' }}>
                  {callType === 'video' ? 'Appel vidéo' : 'Appel audio'} en cours...
                </p>
              </div>
            </div>
          )}
        </div>
        
        {/* Contrôles */}
        <div style={{
          padding: '32px',
          display: 'flex',
          justifyContent: 'center',
          gap: '20px'
        }}>
          <button
            onClick={toggleMute}
            style={{
              width: '56px',
              height: '56px',
              borderRadius: '50%',
              background: isMuted ? '#dc2626' : 'rgba(255,255,255,0.2)',
              border: 'none',
              color: 'white',
              cursor: 'pointer',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center'
            }}
          >
            {isMuted ? <MicOff size={24} /> : <Mic size={24} />}
          </button>
          
          {callType === 'video' && (
            <button
              onClick={toggleVideo}
              style={{
                width: '56px',
                height: '56px',
                borderRadius: '50%',
                background: isVideoOff ? '#dc2626' : 'rgba(255,255,255,0.2)',
                border: 'none',
                color: 'white',
                cursor: 'pointer',
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center'
              }}
            >
              {isVideoOff ? <VideoOff size={24} /> : <Video size={24} />}
            </button>
          )}
          
          <button
            onClick={endCall}
            style={{
              width: '56px',
              height: '56px',
              borderRadius: '50%',
              background: '#dc2626',
              border: 'none',
              color: 'white',
              cursor: 'pointer',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center'
            }}
          >
            <PhoneOff size={24} />
          </button>
        </div>
      </div>
    );
  }
  
  // Interface principale de l'application
  if (screen === 'app' && currentUser) {
    return (
      <div style={{
        minHeight: '100vh',
        background: theme.background,
        fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif',
        color: theme.text,
        display: 'flex',
        flexDirection: 'column'
      }}>
        {/* Header */}
        <div style={{
          background: theme.surface,
          borderBottom: `1px solid ${theme.border}`,
          padding: '16px 24px',
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center',
          position: 'sticky',
          top: 0,
          zIndex: 100
        }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: '16px' }}>
            <Logo size="small" />
          </div>
          
          <div style={{ display: 'flex', alignItems: 'center', gap: '16px' }}>
            <button
              onClick={() => setDarkMode(!darkMode)}
              style={{
                background: 'transparent',
                border: 'none',
                cursor: 'pointer',
                color: theme.text,
                display: 'flex',
                alignItems: 'center'
              }}
            >
              {darkMode ? <Sun size={20} /> : <Moon size={20} />}
            </button>
            
            <div style={{
              display: 'flex',
              alignItems: 'center',
              gap: '12px',
              padding: '8px 16px',
              borderRadius: '24px',
              background: theme.background
            }}>
              <div style={{
                width: '32px',
                height: '32px',
                borderRadius: '50%',
                background: theme.primary,
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                color: 'white',
                fontWeight: 'bold',
                position: 'relative'
              }}>
                {currentUser.prenom[0]}{currentUser.nom[0]}
                {onlineStatus === 'visible' && (
                  <div style={{
                    position: 'absolute',
                    bottom: 0,
                    right: 0,
                    width: '10px',
                    height: '10px',
                    background: '#10b981',
                    border: `2px solid ${theme.surface}`,
                    borderRadius: '50%'
                  }} />
                )}
              </div>
              <div>
                <div style={{ fontSize: '14px', fontWeight: '600' }}>
                  {currentUser.prenom} {currentUser.nom}
                </div>
                <div style={{ fontSize: '12px', color: theme.textSecondary }}>
                  {currentUser.specialite}
                </div>
              </div>
            </div>
          </div>
        </div>
        
        {/* Corps principal */}
        <div style={{ flex: 1, display: 'flex' }}>
          <div style={{ flex: 1, padding: '24px', overflowY: 'auto' }}>
            {activeTab === 'feed' && (
              <div style={{ maxWidth: '800px', margin: '0 auto' }}>
                <h2 style={{ fontSize: '24px', fontWeight: 'bold', marginBottom: '24px' }}>
                  Fil d'actualités
                </h2>
                
                {/* Créer un post */}
                <div style={{
                  background: theme.surface,
                  padding: '24px',
                  borderRadius: '12px',
                  marginBottom: '24px',
                  border: `1px solid ${theme.border}`
                }}>
                  <textarea
                    value={newPost}
                    onChange={(e) => setNewPost(e.target.value)}
                    placeholder="Quoi de neuf ?"
                    style={{
                      width: '100%',
                      minHeight: '100px',
                      padding: '12px',
                      borderRadius: '8px',
                      border: `1px solid ${theme.border}`,
                      background: theme.background,
                      color: theme.text,
                      fontSize: '14px',
                      resize: 'vertical',
                      marginBottom: '12px',
                      boxSizing: 'border-box'
                    }}
                  />
                  
                  <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
                    <div style={{ display: 'flex', gap: '12px' }}>
                      <button style={{
                        background: 'transparent',
                        border: 'none',
                        color: theme.textSecondary,
                        cursor: 'pointer',
                        display: 'flex',
                        alignItems: 'center',
                        gap: '6px'
                      }}>
                        <ImageIcon size={20} /> Photo
                      </button>
